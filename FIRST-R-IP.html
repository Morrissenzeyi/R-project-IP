remove_outliers <- function(df, na.rm = TRUE, ...) {
    qnt <- quantile(df, probs=c(.25, .75), na.rm = na.rm, ...)
    H <- 1.5 * IQR(df, na.rm = na.rm)
    y <- df
    y[df < (qnt[1] - H)] <- NA
    y[df > (qnt[2] + H)] <- NA
    y
}
